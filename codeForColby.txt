
        public void AutoRefreshGrade(ObjectOutputStream objectOutputStream) {
            final ArrayList[] temp = new ArrayList[]{grades};
            Timer timer = new Timer();
            timer.scheduleAtFixedRate(new TimerTask() {
                public void run() {
                    if(!temp[0].equals(grades)) {
                        try {
                            objectOutputStream.writeUTF("RefreshGrade");
                            temp[0] = grades;
                        } catch (IOException e) {
                            e.printStackTrace();
                        }
                    }
                }
            }, 1000, 1000);
        }
